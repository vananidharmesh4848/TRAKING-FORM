function submitForms() { var msgEl = document.getElementById('sub-msg'); msgEl.innerText = "⏳ Saving to Google Sheet..."; var promises = []; var myCats = db.categories.filter(c => c.department.toLowerCase() === activeManager.department.toLowerCase()); myCats.forEach(cat => { var count = uploadedFilesMap[cat.id] ? uploadedFilesMap[cat.id].length : 0; var payload = { action: "submitForm", managerId: activeManager.id, shift: currentShift, date: document.getElementById('sub-date').value, time: document.getElementById('sub-time').value, category: cat.name, filesCount: count, notes: document.getElementById('sub-note').value }; // ગૂગલ સ્ક્રિપ્ટને ડેટા મોકલવો (no-cors વગર) promises.push( fetch(googleWebAppUrl, { method: "POST", body: JSON.stringify(payload) }) ); }); Promise.all(promises) .then(() => { msgEl.innerText = "✅ All data submitted directly to Google Sheet!"; setTimeout(() => { uploadedFilesMap = {}; document.getElementById('sub-note').value = ""; msgEl.innerText = ""; fetchLiveData(); }, 2000); }) .catch(err => { msgEl.innerText = "❌ Error saving data!"; console.error(err); }); } function addMgr() { var payload = { action: "addManager", id: document.getElementById('nm-id').value.toUpperCase().trim(), name: document.getElementById('nm-nm').value.trim(), department: document.getElementById('nm-dept').value, password: document.getElementById('nm-pw').value.trim() }; if(!payload.id || !payload.name || !payload.password) { alert("Please fill all fields"); return; } fetch(googleWebAppUrl, { method: "POST", body: JSON.stringify(payload) }) .then(res => res.json()) .then(data => { alert("Manager Saved to Sheet!"); fetchLiveData(); }) .catch(err => alert("Error: Data not saved!")); } function addDept() { var name = document.getElementById('nd-nm').value.trim(); if(!name) return; var payload = { action: "addDept", name: name }; fetch(googleWebAppUrl, { method: "POST", body: JSON.stringify(payload) }) .then(res => res.json()) .then(data => { alert("Dept Saved to Sheet!"); document.getElementById('nd-nm').value = ""; fetchLiveData(); }); } function addCat() { var name = document.getElementById('nc-nm').value.trim(); var dept = document.getElementById('nc-dept').value; if(!name) return; var payload = { action: "addCat", id: "cat_" + Date.now(), name: name, department: dept }; fetch(googleWebAppUrl, { method: "POST", body: JSON.stringify(payload) }) .then(res => res.json()) .then(data => { alert("Category Saved to Sheet!"); document.getElementById('nc-nm').value = ""; fetchLiveData(); }); }
